project(SaintCore)

        googletest
        GIT_REPOSITORY https://github.com/google/googletest.git
        GIT_TAG v1.14.0
)

        SaintCoreTests
        src/functions.cpp
        src/model.cpp
        src/tensor.cpp
        tests/test_functions.cpp
        tests/test_tensor.cpp
)

target_include_directories(SaintCoreTests PRIVATE include)
